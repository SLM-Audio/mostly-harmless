30 template <DatabaseStorageType T>
31 auto databaseQueryCallback(void* ud, int count, char** data, char** /*columns*/) -> int {
32 auto* result = static_cast<std::optional<T>*>(ud);
33 if (count != 6) {
34 return SQLITE_FAIL;
35 }
36 if constexpr (std::same_as<T, std::string>) {
37 *result = data[Index::TextIndex];
38 } else if constexpr (std::same_as<T, bool>) {
39 *result = static_cast<T>(std::stoi(data[Index::BoolIndex]));
40 } else if constexpr (std::is_integral_v<T>) {
41 *result = static_cast<T>(std::stoi(data[Index::IntIndex]));
42 } else if constexpr (std::same_as<T, float>) {
43 *result = std::stof(data[Index::FloatIndex]);
44 } else if constexpr (std::same_as<T, double>) {
45 *result = std::stod(data[Index::DoubleIndex]);
46 }
47 return SQLITE_OK;
48 }

124 std::string textValue{};

131 std::string updateStr;

133 props.textValue = toSet;

135 } else if constexpr (std::same_as<T, bool>) {
136 props.boolValue = toSet;
137 updateStr = fmt::format("BOOL_VALUE = {}", props.boolValue);
138 } else if constexpr (std::is_integral_v<T>) {
139 props.intValue = static_cast<int>(toSet);
140 updateStr = fmt::format("INT_VALUE = {}", props.intValue);
141 } else if constexpr (std::same_as<T, float>) {
142 props.floatValue = toSet;
143 updateStr = fmt::format("FLOAT_VALUE = {}", props.floatValue);
144 } else if constexpr (std::same_as<T, double>) {
145 props.doubleValue = toSet;
146 updateStr = fmt::format("DOUBLE_VALUE = {}", props.doubleValue);
147 }
148
149 std::stringstream sstream;
150 sstream << "INSERT INTO DATA (NAME, TEXT_VALUE, BOOL_VALUE, INT_VALUE, FLOAT_VALUE, DOUBLE_VALUE)\n";
151 sstream << " VALUES ( \"" << name << "\", \"" << props.textValue << "\", " << props.boolValue << "," << props.intValue << ", " << props.floatValue << ", " << props.doubleValue << " )\n";
152 sstream << " ON CONFLICT(NAME) DO UPDATE SET " << updateStr << ";";
153 const auto execResult = sqlite3_exec(m_databaseHandle, sstream.str().c_str(), nullptr, nullptr, nullptr);
154 if (execResult != SQLITE_OK) {
