29 template <DatabaseStorageType T>
30 auto databaseQueryCallback(void* ud, int count, char** data, char** /*columns*/) -> int {
31 auto* result = static_cast<std::optional<T>*>(ud);
32 if (count != 6) {
33 return SQLITE_FAIL;
34 }
35 if constexpr (std::same_as<T, std::string>) {
36 *result = data[Index::TextIndex];
37 } else if constexpr (std::same_as<T, bool>) {
38 *result = static_cast<T>(std::stoi(data[Index::BoolIndex]));
39 } else if constexpr (std::is_integral_v<T>) {
40 *result = static_cast<T>(std::stoi(data[Index::IntIndex]));
41 } else if constexpr (std::same_as<T, float>) {
42 *result = std::stof(data[Index::FloatIndex]);
43 } else if constexpr (std::same_as<T, double>) {
44 *result = std::stod(data[Index::DoubleIndex]);
45 }
46 return SQLITE_OK;
47 }

74 DatabaseState(Private, const std::filesystem::path& location, const std::vector<std::pair<std::string, DatabaseValueVariant>>& initialValues) {
75 const auto checkResult = [](int response) -> void {
76 if (response != SQLITE_OK) {
77 throw std::exception{};
78 }
79 };
80
81 // Try open existing
82 if (sqlite3_open_v2(location.string().c_str(), &m_databaseHandle, SQLITE_OPEN_READWRITE, nullptr) != SQLITE_OK) {
83 checkResult(sqlite3_open(location.string().c_str(), &m_databaseHandle)); // Didn't exist, try create
84 }
85 // Enable WAL
86 checkResult(sqlite3_exec(m_databaseHandle, "PRAGMA journal_mode=WAL", nullptr, nullptr, nullptr));
87 // Create Table if not present
88 checkResult(sqlite3_exec(m_databaseHandle,
89 "CREATE TABLE IF NOT EXISTS DATA (NAME text UNIQUE, TEXT_VALUE text, BOOL_VALUE bool, INT_VALUE int, FLOAT_VALUE float, DOUBLE_VALUE double);",
90 nullptr,
91 nullptr,
92 nullptr));
93 // Populate with initial values, if key is present already, skip set
94 for (const auto& [key, value] : initialValues) {
95 std::visit([this, &key](auto&& arg) {
96 using T = std::decay_t<decltype(arg)>;
97 if (get<T>(key)) {
98 return;
99 }
100 set(key, std::forward<decltype(arg)>(arg));
101 },
102 value);
103 }
104 }

172 template <DatabaseStorageType T>
173 auto set(std::string_view name, const T& toSet) -> void {
174 struct {
175 std::string textValue{};
176 bool boolValue{ false };
177 int intValue{ 0 };
178 float floatValue{ 0.0f };
179 double doubleValue{ 0.0 };
180 } props;
181
182 std::string updateStr;
183 if constexpr (std::same_as<T, std::string>) {
184 props.textValue = toSet;
185 updateStr = fmt::format("TEXT_VALUE = \"{}\"", props.textValue);
186 } else if constexpr (std::same_as<T, bool>) {
187 props.boolValue = toSet;
188 updateStr = fmt::format("BOOL_VALUE = {}", props.boolValue);
189 } else if constexpr (std::is_integral_v<T>) {
190 props.intValue = static_cast<int>(toSet);
191 updateStr = fmt::format("INT_VALUE = {}", props.intValue);
192 } else if constexpr (std::same_as<T, float>) {
193 props.floatValue = toSet;
194 updateStr = fmt::format("FLOAT_VALUE = {}", props.floatValue);
195 } else if constexpr (std::same_as<T, double>) {
196 props.doubleValue = toSet;
197 updateStr = fmt::format("DOUBLE_VALUE = {}", props.doubleValue);
198 }
199
200 std::stringstream sstream;
201 sstream << "INSERT INTO DATA (NAME, TEXT_VALUE, BOOL_VALUE, INT_VALUE, FLOAT_VALUE, DOUBLE_VALUE)\n";
202 sstream << " VALUES ( \"" << name << "\", \"" << props.textValue << "\", " << props.boolValue << "," << props.intValue << ", " << props.floatValue << ", " << props.doubleValue << " )\n";
203 sstream << " ON CONFLICT(NAME) DO UPDATE SET " << updateStr << ";";
204 const auto execResult = sqlite3_exec(m_databaseHandle, sstream.str().c_str(), nullptr, nullptr, nullptr);
205 if (execResult != SQLITE_OK) {
206 MH_LOG(sqlite3_errmsg(m_databaseHandle));
207 assert(false);
208 }

218 [[nodiscard]] auto get(std::string_view name) -> std::optional<T> {
219 std::optional<T> result{};
220 const auto execStr = fmt::format("SELECT * FROM DATA WHERE NAME = \"{}\"", name);
221 const auto getRes = sqlite3_exec(m_databaseHandle,
222 execStr.c_str(),
223 databaseQueryCallback<T>,
224 static_cast<void*>(&result),
225 nullptr);
226 if (getRes != SQLITE_OK) {
227 MH_LOG(sqlite3_errmsg(m_databaseHandle));
228 assert(false);
229 }
