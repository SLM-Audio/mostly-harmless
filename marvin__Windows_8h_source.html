51 template <FloatType SampleType>
52 [[nodiscard]] SampleType tukey(SampleType n, SampleType N, SampleType alpha) {
53 // 0.5 * (1 - cos(2pin/alphaN)), 0 <= n <= aN / 2
54 // 1, aN/2 <= n <= N / 2
55 // w[N - n] = w[n], 0 <= n <= N / 2
56 const auto alphaN = alpha * N;
57 const auto NOverTwo = N / static_cast<SampleType>(2.0);
58 const auto aNOverTwo = alphaN / static_cast<SampleType>(2.0);
59 if (n >= 0 && n <= aNOverTwo) {
60 // 0.5 * (1 - cos(2pin / alphaN))
61 constexpr static auto twoPi = static_cast<SampleType>(2.0) * std::numbers::pi_v<SampleType>;
62 const auto cosine = std::cos((twoPi * n) / alphaN);
63 const auto res = static_cast<SampleType>(0.5) * (static_cast<SampleType>(1.0 - cosine));
64 return res;
65 } else if (n > aNOverTwo && n <= NOverTwo) {
66 return static_cast<SampleType>(1.0);
67 } else {
68 const auto reverseIndex = N - n;
69 return tukey(reverseIndex, N, alpha);
70 }
71 }

79 template <FloatType SampleType>
80 [[nodiscard]] SampleType blackmanHarris(SampleType n, SampleType N) {
81 constexpr static auto a0{ 0.35875 };
82 constexpr static auto a1{ 0.48829 };
83 constexpr static auto a2{ 0.14128 };
84 constexpr static auto a3{ 0.01168 };
85 constexpr static auto pi{ std::numbers::pi_v<double> };
86 constexpr static auto twoPi{ pi * 2.0 };
87 constexpr static auto fourPi{ pi * 4.0 };
88 constexpr static auto sixPi{ pi * 6.0 };
89 const auto position = static_cast<double>(n) / static_cast<double>(N);
90 const auto a1Term = a1 * std::cos(position * twoPi);
91 const auto a2Term = a2 * std::cos(position * fourPi);
92 const auto a3Term = a3 * std::cos(position * sixPi);
93 const auto result = a0 - a1Term + a2Term - a3Term;
94 return static_cast<SampleType>(result);
95 }

107 template <FloatType SampleType>
108 [[nodiscard]] SampleType cosineSum(SampleType n, SampleType N, SampleType alpha) {
109 // w[n] = a0 - (1 - a0) * cos(2pi * n/N)
110 const auto a0 = alpha;
111 const auto a1 = static_cast<SampleType>(1.0) - alpha;
112 constexpr static auto twoPi = std::numbers::pi_v<SampleType> * static_cast<SampleType>(2.0);
113 const auto ratio = n / N;
114 const auto cosine = std::cos(twoPi * ratio);
115 const auto a1Scaled = a1 * cosine;
116 const auto res = a0 - a1Scaled;
117 return res;
118 }
